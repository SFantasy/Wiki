Flexible boxes

// Build flexible box
display: flex;
display: -webkit-flex;
display: -moz-flex;
display: -ms-flex;
display: box;
display: flexbox;
display: -webkit-box;
display: -moz-box;

// Flex direction
flex-direction: row;
-webkit-flex-direction: row;
//
flex-grow: 1;
-webkit-flex-row: 1;
